-----------------------------------------------
two routines (functions) non-threaded 
-----------------------------------------------
#include<iostream>
#include<cstdlib>
#include<windows.h>
using namespace std;

void parentSayHi() {
	for(int I=1; I <= 6; I++){
	Sleep(300);
	cout << "Hi from parent" << endl;
	}
	Sleep(300);
}
void childSayHello() {
	for (int I = 1; I <= 6; I++) {
		Sleep(300);
		cout << "Hello from child" << endl;
	}
	Sleep(300);
}
int main() {
	cout << "Waiting...." << endl;
	childSayHello();
	parentSayHi();
	return EXIT_SUCCESS;
}



-----------------------------------------------
two routines (functions) threaded 
-----------------------------------------------
#include<iostream>
#include<cstdlib>
#include<windows.h>
#include<thread>
using namespace std;

void parentSayHi() {
	for(int I=1; I <= 6; I++){
	Sleep(300);
	cout << "Hi from parent" << endl;
	}
	Sleep(300);
}
void childSayHello() {
	for (int I = 1; I <= 6; I++) {
		Sleep(300);
		cout << "Hello from child" << endl;
	}
	Sleep(300);
}
int main() {
	cout << "Non-threaded...." << endl;
    childSayHello();
    parentSayHi();
    
    cout << "Threaded....manager-worker" << endl;
    thread child(childSayHello); //worker 
	parentSayHi();//manager 
    child.join();//synchronization 
    
    
	return EXIT_SUCCESS;
}
